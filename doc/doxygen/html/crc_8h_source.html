353 template <typename CRCType, crcpp_uint16 CRCWidth>
354 inline void CRC::Table<CRCType, CRCWidth>::InitTable()
355 {
356  // For masking off the bits for the CRC (in the event that the number of bits in CRCType is larger than CRCWidth)
357  static crcpp_constexpr CRCType BIT_MASK((CRCType(1) << (CRCWidth - CRCType(1))) |
358  ((CRCType(1) << (CRCWidth - CRCType(1))) - CRCType(1)));
359 
360  // The conditional expression is used to avoid a -Wshift-count-overflow warning.
361  static crcpp_constexpr CRCType SHIFT((CHAR_BIT >= CRCWidth) ? static_cast<CRCType>(CHAR_BIT - CRCWidth) : 0);
362 
363  CRCType crc;
364  unsigned char byte = 0;
365 
366  // Loop over each dividend (each possible number storable in an unsigned char)
367  do
368  {
369  crc = CRC::CalculateRemainder<CRCType, CRCWidth>(&byte, sizeof(byte), parameters, CRCType(0));
370 
371  // This mask might not be necessary; all unit tests pass with this line commented out,
372  // but that might just be a coincidence based on the CRC parameters used for testing.
373  // In any case, this is harmless to leave in and only adds a single machine instruction per loop iteration.
374  crc &= BIT_MASK;
375 
376  if (!parameters.reflectInput && CRCWidth < CHAR_BIT)
377  {
378  // Undo the special operation at the end of the CalculateRemainder()
379  // function for non-reflected CRCs < CHAR_BIT.
380  crc = static_cast<CRCType>(crc << SHIFT);
381  }
382 
383  table[byte] = crc;
384  }
385  while (++byte);
386 }

484 template <typename CRCType, crcpp_uint16 CRCWidth>
485 inline CRCType CRC::CalculateBits(const void * data, crcpp_size size, const Parameters<CRCType, CRCWidth> & parameters)
486 {
487  CRCType remainder = parameters.initialValue;
488 
489  // Calculate the remainder on a whole number of bytes first, then call
490  // a special-case function for the remaining bits.
491  crcpp_size wholeNumberOfBytes = size / CHAR_BIT;
492  if (wholeNumberOfBytes > 0)
493  {
494  remainder = CalculateRemainder(data, wholeNumberOfBytes, parameters, remainder);
495  }
496 
497  crcpp_size remainingNumberOfBits = size % CHAR_BIT;
498  if (remainingNumberOfBits != 0)
499  {
500  unsigned char lastByte = *(reinterpret_cast<const unsigned char *>(data) + wholeNumberOfBytes);
501  remainder = CalculateRemainderBits(lastByte, remainingNumberOfBits, parameters, remainder);
502  }
503 
504  // No need to mask the remainder here; the mask will be applied in the Finalize() function.
505 
506  return Finalize<CRCType, CRCWidth>(remainder, parameters.finalXOR, parameters.reflectInput != parameters.reflectOutput);
507 }

626 inline IntegerType CRC::Reflect(IntegerType value, crcpp_uint16 numBits)
627 {
628  IntegerType reversedValue(0);
629 
630  for (crcpp_uint16 i = 0; i < numBits; ++i)
631  {

648 template <typename CRCType, crcpp_uint16 CRCWidth>
649 inline CRCType CRC::Finalize(CRCType remainder, CRCType finalXOR, bool reflectOutput)
650 {
651  // For masking off the bits for the CRC (in the event that the number of bits in CRCType is larger than CRCWidth)
652  static crcpp_constexpr CRCType BIT_MASK = (CRCType(1) << (CRCWidth - CRCType(1))) |
653  ((CRCType(1) << (CRCWidth - CRCType(1))) - CRCType(1));
654 
655  if (reflectOutput)
656  {
657  remainder = Reflect(remainder, CRCWidth);
658  }
659 
660  return (remainder ^ finalXOR) & BIT_MASK;
661 }

680 template <typename CRCType, crcpp_uint16 CRCWidth>
681 inline CRCType CRC::UndoFinalize(CRCType crc, CRCType finalXOR, bool reflectOutput)
682 {
683  // For masking off the bits for the CRC (in the event that the number of bits in CRCType is larger than CRCWidth)
684  static crcpp_constexpr CRCType BIT_MASK = (CRCType(1) << (CRCWidth - CRCType(1))) |

687  crc = (crc & BIT_MASK) ^ finalXOR;

707 template <typename CRCType, crcpp_uint16 CRCWidth>
708 inline CRCType CRC::CalculateRemainder(const void * data, crcpp_size size, const Parameters<CRCType, CRCWidth> & parameters, CRCType remainder)
709 {
710 #ifdef CRCPP_USE_CPP11
711  // This static_assert is put here because this function will always be compiled in no matter what
712  // the template parameters are and whether or not a table lookup or bit-by-bit algorithm is used.
713  static_assert(::std::numeric_limits<CRCType>::digits >= CRCWidth, "CRCType is too small to contain a CRC of width CRCWidth.");
714 #else
715  // Catching this compile-time error is very important. Sadly, the compiler error will be very cryptic, but it's
716  // better than nothing.
717  enum { static_assert_failed_CRCType_is_too_small_to_contain_a_CRC_of_width_CRCWidth = 1 / (::std::numeric_limits<CRCType>::digits >= CRCWidth ? 1 : 0) };
718 #endif
719 
720  const unsigned char * current = reinterpret_cast<const unsigned char *>(data);
721 
722  // Slightly different implementations based on the parameters. The current implementations try to eliminate as much
723  // computation from the inner loop (looping over each bit) as possible.
724  if (parameters.reflectInput)
725  {
726  CRCType polynomial = CRC::Reflect(parameters.polynomial, CRCWidth);
727  while (size--)
728  {
729  remainder = static_cast<CRCType>(remainder ^ *current++);
730 
731  // An optimizing compiler might choose to unroll this loop.
732  for (crcpp_size i = 0; i < CHAR_BIT; ++i)
733  {
734 #ifdef CRCPP_BRANCHLESS
735  // Clever way to avoid a branch at the expense of a multiplication. This code is equivalent to the following:
736  // if (remainder & 1)
737  // remainder = (remainder >> 1) ^ polynomial;
738  // else
739  // remainder >>= 1;
740  remainder = static_cast<CRCType>((remainder >> 1) ^ ((remainder & 1) * polynomial));
741 #else
742  remainder = static_cast<CRCType>((remainder & 1) ? ((remainder >> 1) ^ polynomial) : (remainder >> 1));
743 #endif
744  }
745  }
746  }
747  else if (CRCWidth >= CHAR_BIT)
748  {
749  static crcpp_constexpr CRCType CRC_WIDTH_MINUS_ONE(CRCWidth - CRCType(1));
750 #ifndef CRCPP_BRANCHLESS
751  static crcpp_constexpr CRCType CRC_HIGHEST_BIT_MASK(CRCType(1) << CRC_WIDTH_MINUS_ONE);
752 #endif
753  // The conditional expression is used to avoid a -Wshift-count-overflow warning.
754  static crcpp_constexpr CRCType SHIFT((CRCWidth >= CHAR_BIT) ? static_cast<CRCType>(CRCWidth - CHAR_BIT) : 0);
755 
756  while (size--)
757  {
758  remainder = static_cast<CRCType>(remainder ^ (static_cast<CRCType>(*current++) << SHIFT));
759 
760  // An optimizing compiler might choose to unroll this loop.
761  for (crcpp_size i = 0; i < CHAR_BIT; ++i)
762  {
763 #ifdef CRCPP_BRANCHLESS
764  // Clever way to avoid a branch at the expense of a multiplication. This code is equivalent to the following:
765  // if (remainder & CRC_HIGHEST_BIT_MASK)
766  // remainder = (remainder << 1) ^ parameters.polynomial;
767  // else
768  // remainder <<= 1;
769  remainder = static_cast<CRCType>((remainder << 1) ^ (((remainder >> CRC_WIDTH_MINUS_ONE) & 1) * parameters.polynomial));
770 #else
771  remainder = static_cast<CRCType>((remainder & CRC_HIGHEST_BIT_MASK) ? ((remainder << 1) ^ parameters.polynomial) : (remainder << 1));
772 #endif
773  }
774  }
775  }
776  else
777  {
778  static crcpp_constexpr CRCType CHAR_BIT_MINUS_ONE(CHAR_BIT - 1);
779 #ifndef CRCPP_BRANCHLESS
780  static crcpp_constexpr CRCType CHAR_BIT_HIGHEST_BIT_MASK(CRCType(1) << CHAR_BIT_MINUS_ONE);
781 #endif
782  // The conditional expression is used to avoid a -Wshift-count-overflow warning.
783  static crcpp_constexpr CRCType SHIFT((CHAR_BIT >= CRCWidth) ? static_cast<CRCType>(CHAR_BIT - CRCWidth) : 0);
784 
785  CRCType polynomial = static_cast<CRCType>(parameters.polynomial << SHIFT);
786  remainder = static_cast<CRCType>(remainder << SHIFT);
787 
788  while (size--)
789  {
790  remainder = static_cast<CRCType>(remainder ^ *current++);
791 
792  // An optimizing compiler might choose to unroll this loop.
793  for (crcpp_size i = 0; i < CHAR_BIT; ++i)
794  {
795 #ifdef CRCPP_BRANCHLESS
796  // Clever way to avoid a branch at the expense of a multiplication. This code is equivalent to the following:
797  // if (remainder & CHAR_BIT_HIGHEST_BIT_MASK)
798  // remainder = (remainder << 1) ^ polynomial;
799  // else
800  // remainder <<= 1;
801  remainder = static_cast<CRCType>((remainder << 1) ^ (((remainder >> CHAR_BIT_MINUS_ONE) & 1) * polynomial));
802 #else
803  remainder = static_cast<CRCType>((remainder & CHAR_BIT_HIGHEST_BIT_MASK) ? ((remainder << 1) ^ polynomial) : (remainder << 1));
804 #endif
805  }
806  }
807 
808  remainder = static_cast<CRCType>(remainder >> SHIFT);
809  }
810 
811  return remainder;
812 }

824 template <typename CRCType, crcpp_uint16 CRCWidth>
825 inline CRCType CRC::CalculateRemainder(const void * data, crcpp_size size, const Table<CRCType, CRCWidth> & lookupTable, CRCType remainder)
826 {
827  const unsigned char * current = reinterpret_cast<const unsigned char *>(data);
828 
829  if (lookupTable.GetParameters().reflectInput)
830  {
831  while (size--)
832  {
833 #if defined(WIN32) || defined(_WIN32) || defined(WINCE)
834  // Disable warning about data loss when doing (remainder >> CHAR_BIT) when
835  // remainder is one byte long. The algorithm is still correct in this case,
836  // though it's possible that one additional machine instruction will be executed.
837 # pragma warning (push)
838 # pragma warning (disable : 4333)
839 #endif
840  remainder = static_cast<CRCType>((remainder >> CHAR_BIT) ^ lookupTable[static_cast<unsigned char>(remainder ^ *current++)]);
841 #if defined(WIN32) || defined(_WIN32) || defined(WINCE)
842 # pragma warning (pop)
843 #endif
844  }
845  }
846  else if (CRCWidth >= CHAR_BIT)
847  {
848  // The conditional expression is used to avoid a -Wshift-count-overflow warning.
849  static crcpp_constexpr CRCType SHIFT((CRCWidth >= CHAR_BIT) ? static_cast<CRCType>(CRCWidth - CHAR_BIT) : 0);
850 
851  while (size--)
852  {
853  remainder = static_cast<CRCType>((remainder << CHAR_BIT) ^ lookupTable[static_cast<unsigned char>((remainder >> SHIFT) ^ *current++)]);
854  }
855  }
856  else
857  {
858  // The conditional expression is used to avoid a -Wshift-count-overflow warning.
859  static crcpp_constexpr CRCType SHIFT((CHAR_BIT >= CRCWidth) ? static_cast<CRCType>(CHAR_BIT - CRCWidth) : 0);
860 
861  remainder = static_cast<CRCType>(remainder << SHIFT);
862 
863  while (size--)
864  {
865  // Note: no need to mask here since remainder is guaranteed to fit in a single byte.
866  remainder = lookupTable[static_cast<unsigned char>(remainder ^ *current++)];
867  }
868 
869  remainder = static_cast<CRCType>(remainder >> SHIFT);
870  }
871 
872  return remainder;
873 }

875 template <typename CRCType, crcpp_uint16 CRCWidth>
876 inline CRCType CRC::CalculateRemainderBits(unsigned char byte, crcpp_size numBits, const Parameters<CRCType, CRCWidth> & parameters, CRCType remainder)
877 {
878  // Slightly different implementations based on the parameters. The current implementations try to eliminate as much
879  // computation from the inner loop (looping over each bit) as possible.
880  if (parameters.reflectInput)
881  {
882  CRCType polynomial = CRC::Reflect(parameters.polynomial, CRCWidth);
883  remainder = static_cast<CRCType>(remainder ^ byte);
884

886  for (crcpp_size i = 0; i < numBits; ++i)
887  {

890  // if (remainder & 1)
891  // remainder = (remainder >> 1) ^ polynomial;
892  // else
893  // remainder >>= 1;
894  remainder = static_cast<CRCType>((remainder >> 1) ^ ((remainder & 1) * polynomial));
895 #else
896  remainder = static_cast<CRCType>((remainder & 1) ? ((remainder >> 1) ^ polynomial) : (remainder >> 1));
897 #endif
898  }
899  }

903 #ifndef CRCPP_BRANCHLESS
904  static crcpp_constexpr CRCType CRC_HIGHEST_BIT_MASK(CRCType(1) << CRC_WIDTH_MINUS_ONE);

907  static crcpp_constexpr CRCType SHIFT((CRCWidth >= CHAR_BIT) ? static_cast<CRCType>(CRCWidth - CHAR_BIT) : 0);
908 
909  remainder = static_cast<CRCType>(remainder ^ (static_cast<CRCType>(byte) << SHIFT));
910 
911  // An optimizing compiler might choose to unroll this loop.
912  for (crcpp_size i = 0; i < numBits; ++i)
913  {
914 #ifdef CRCPP_BRANCHLESS
915  // Clever way to avoid a branch at the expense of a multiplication. This code is equivalent to the following:
916  // if (remainder & CRC_HIGHEST_BIT_MASK)
917  // remainder = (remainder << 1) ^ parameters.polynomial;
918  // else
919  // remainder <<= 1;
920  remainder = static_cast<CRCType>((remainder << 1) ^ (((remainder >> CRC_WIDTH_MINUS_ONE) & 1) * parameters.polynomial));
921 #else

923 #endif
924  }
925  }

929 #ifndef CRCPP_BRANCHLESS
930  static crcpp_constexpr CRCType CHAR_BIT_HIGHEST_BIT_MASK(CRCType(1) << CHAR_BIT_MINUS_ONE);
931 #endif
932  // The conditional expression is used to avoid a -Wshift-count-overflow warning.
933  static crcpp_constexpr CRCType SHIFT((CHAR_BIT >= CRCWidth) ? static_cast<CRCType>(CHAR_BIT - CRCWidth) : 0);
934

936  remainder = static_cast<CRCType>((remainder << SHIFT) ^ byte);
937 
938  // An optimizing compiler might choose to unroll this loop.
939  for (crcpp_size i = 0; i < numBits; ++i)
940  {
941 #ifdef CRCPP_BRANCHLESS
942  // Clever way to avoid a branch at the expense of a multiplication. This code is equivalent to the following:
943  // if (remainder & CHAR_BIT_HIGHEST_BIT_MASK)
944  // remainder = (remainder << 1) ^ polynomial;
945  // else
946  // remainder <<= 1;
947  remainder = static_cast<CRCType>((remainder << 1) ^ (((remainder >> CHAR_BIT_MINUS_ONE) & 1) * polynomial));

950 #endif
